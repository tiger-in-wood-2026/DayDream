```python
import random

class Game2048:
    def __init__(self):
        self.size = 4
        self.board = [[0] * self.size for _ in range(self.size)]
        self.score = 0
        self.moved = False
        self._add_random_tile()
        self._add_random_tile()

    def _add_random_tile(self):
        empty = [(r, c) for r in range(self.size) for c in range(self.size) if self.board[r][c] == 0]
        if not empty:
            return
        r, c = random.choice(empty)
        self.board[r][c] = 2 if random.random() < 0.9 else 4

    def _compress(self, row):
        """Shift non-zero elements to the left."""
        new_row = [v for v in row if v != 0]
        new_row += [0] * (self.size - len(new_row))
        return new_row

    def _merge(self, row):
        """Merge adjacent equal elements (left to right)."""
        new_row = []
        merged = [False] * self.size
        i = 0
        while i < self.size:
            if i + 1 < self.size and row[i] == row[i + 1] and not merged[i]:
                val = row[i] * 2
                new_row.append(val)
                self.score += val
                merged[i] = True
                merged[i + 1] = True
                i += 2
            else:
                new_row.append(row[i])
                i += 1
        new_row += [0] * (self.size - len(new_row))
        return new_row

    def _transpose(self):
        self.board = [[self.board[r][c] for r in range(self.size)] for c in range(self.size)]

    def _reverse_rows(self):
        self.board = [row[::-1] for row in self.board]

    def move(self, direction):
        """direction: 'left', 'right', 'up', 'down'"""
        old_board = [row[:] for row in self.board]

        if direction == 'left':
            for i in range(self.size):
                self.board[i] = self._compress(self.board[i])
                self.board[i] = self._merge(self.board[i])
        elif direction == 'right':
            for i in range(self.size):
                self.board[i] = self._compress(self.board[i][::-1])
                self.board[i] = self._merge(self.board[i])
        elif direction == 'up':
            self._transpose()
            for i in range(self.size):
                self.board[i] = self._compress(self.board[i])
                self.board[i] = self._merge(self.board[i])
            self._transpose()
        elif direction == 'down':
            self._transpose()
            for i in range(self.size):
                self.board[i] = self._compress(self.board[i][::-1])
                self.board[i] = self._merge(self.board[i])
            self._transpose()

        # Check if board changed
        self.moved = (self.board != old_board)
        if self.moved:
            self._add_random_tile()

    def is_game_over(self):
        for r in range(self.size):
            for c in range(self.size):
                if self.board[r][c] == 0:
                    return False
                if c + 1 < self.size and self.board[r][c] == self.board[r][c + 1]:
                    return False
                if r + 1 < self.size and self.board[r][c] == self.board[r + 1][c]:
                    return False
        return True

    def has_won(self):
        for r in range(self.size):
            for c in range(self.size):
                if self.board[r][c] == 2048:
                    return True
        return False

    def reset(self):
        self.__init__()

    def get_board(self):
        return [row[:] for row in self.board]
```

```python
def draw_board(game, img_size=(600, 700)):
    """Render the 2048 game board as an OpenCV image."""
    import numpy as np
    # Background
    board_img = np.zeros((img_size[1], img_size[0], 3), dtype=np.uint8)
    board_img[:] = (187, 173, 160)  # board background

    # Title area
    cv2.putText(board_img, "2048", (20, 60),
                cv2.FONT_HERSHEY_DUPLEX, 2.0, (119, 110, 101), 2)
    cv2.putText(board_img, f"SCORE: {game.score}",
                (img_size[0] - 200, 40),
                cv2.FONT_HERSHEY_SIMPLEX, 0.7, (119, 110, 101), 2)

    # Calculate board offset to center it
    board_x = (img_size[0] - BOARD_SIZE) // 2
    board_y = 100

    # Draw grid and tiles
    for r in range(4):
        for c in range(4):
            x = board_x + c * (CELL_SIZE + GAP) + GAP
            y = board_y + r * (CELL_SIZE + GAP) + GAP
            val = game.board[r][c]
            color = TILE_COLORS.get(val, (60, 58, 50))
            cv2.rectangle(board_img, (x, y), (x + CELL_SIZE, y + CELL_SIZE),
                          color, -1)

            if val != 0:
                text = str(val)
                font_scale = 1.5 if val < 100 else 1.2 if val < 1000 else 0.9
                text_color = TILE_TEXT_COLOR.get(val, (255, 255, 255))
                (tw, th), _ = cv2.getTextSize(text, cv2.FONT_HERSHEY_BOLD,
                                              font_scale, 2)
                tx = x + (CELL_SIZE - tw) // 2
                ty = y + (CELL_SIZE + th) // 2
                cv2.putText(board_img, text, (tx, ty),
                            cv2.FONT_HERSHEY_BOLD, font_scale, text_color, 2)

    # Game over overlay
    if game.is_game_over():
        overlay = board_img.copy()
        cv2.rectangle(overlay, (board_x, board_y),
                      (board_x + BOARD_SIZE, board_y + BOARD_SIZE),
                      (0, 0, 0), -1)
        cv2.addWeighted(overlay, 0.5, board_img, 0.5, 0, board_img)
        cv2.putText(board_img, "GAME OVER", (img_size[0]//2 - 120, img_size[1]//2),
                    cv2.FONT_HERSHEY_DUPLEX, 1.5, (255, 255, 255), 3)

    return board_img
```

```python
# --- Hand Detection ---

def detect_hand(frame):
    """
    Detect hand and find fingertip using skin color + contour analysis.
    Returns (fingertip_x, fingertip_y) or None if no hand found.
    """
    import cv2
    import numpy as np

    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
    # Skin color range (tuned for East Asian skin tones)
    lower_skin = np.array([0, 30, 50], dtype=np.uint8)
    upper_skin = np.array([30, 150, 255], dtype=np.uint8)
    mask = cv2.inRange(hsv, lower_skin, upper_skin)

    # Noise removal
    mask = cv2.GaussianBlur(mask, (5, 5), 0)
    kernel = np.ones((5, 5), np.uint8)
    mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
    mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)

    # Find contours
    contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    if not contours:
        return None, mask

    # Largest contour = hand
    hand_contour = max(contours, key=cv2.contourArea)
    if cv2.contourArea(hand_contour) < 1000:
        return None, mask

    # Convex hull
    hull = cv2.convexHull(hand_contour, returnPoints=True)

    # Find fingertip: the highest point on the hull (minimum y)
    # Also consider convexity defects to find extended fingers
    fingertip = None
    if len(hull) > 0:
        # The tip is the point with smallest y (topmost)
        tip_idx = np.argmin(hull[:, 0, 1])
        fingertip = tuple(hull[tip_idx][0])

    return fingertip, mask
```

```python
# --- Gesture Recognition ---

class GestureController:
    def __init__(self, swipe_threshold=40, cooldown_ms=500):
        self.track_points = []
        self.max_track = 10
        self.swipe_threshold = swipe_threshold
        self.last_trigger_time = 0
        self.cooldown_ms = cooldown_ms
        self.last_direction = None

    def update(self, fingertip):
        """Process a new fingertip position. Returns direction or None."""
        import time
        now = time.time() * 1000

        if fingertip is None:
            self.track_points.clear()
            return None

        self.track_points.append(fingertip)
        if len(self.track_points) > self.max_track:
            self.track_points.pop(0)

        if len(self.track_points) < self.max_track:
            return None

        # Cooldown check
        if now - self.last_trigger_time < self.cooldown_ms:
            return None

        # Calculate movement from track history
        first = self.track_points[0]
        last = self.track_points[-1]
        dx = last[0] - first[0]
        dy = last[1] - first[1]

        # Need minimum movement
        if abs(dx) < self.swipe_threshold and abs(dy) < self.swipe_threshold:
            return None

        # Determine direction
        if abs(dx) > abs(dy):
            direction = 'right' if dx > 0 else 'left'
        else:
            direction = 'down' if dy > 0 else 'up'

        self.last_trigger_time = now
        self.last_direction = direction
        return direction

    def draw_track(self, img):
        """Draw finger tracking trail on the image."""
        for i, pt in enumerate(self.track_points):
            alpha = i / len(self.track_points)
            color = (0, int(255 * alpha), int(255 * (1 - alpha)))
            cv2.circle(img, pt, 5, (0, 0, 255), -1)
        if len(self.track_points) > 1:
            for i in range(len(self.track_points) - 1):
                alpha = i / len(self.track_points)
                cv2.line(img, self.track_points[i], self.track_points[i + 1],
                         (0, 255, 0), 2)
```

```python
def main():
    print("[OK] Starting 2048 Gesture Game...")
    game = Game2048()
    gesture = GestureController()

    cap = cv2.VideoCapture(0)
    if not cap.isOpened():
        print("ERROR: Cannot open camera")
        return
    print("[OK] Camera opened")

    # Camera preview size for the PIP window
    cam_width, cam_height = 240, 180

    while True:
        ret, frame = cap.read()
        if not ret:
            print("WARNING: Camera read failed")
            continue

        # Mirror the camera feed
        frame = cv2.flip(frame, 1)

        # Resize for PIP
        cam_small = cv2.resize(frame, (cam_width, cam_height))

        # Hand detection on camera feed
        fingertip, mask = detect_hand(frame)

        # Gesture recognition
        direction = gesture.update(fingertip)
        if direction:
            game.move(direction)
            print(f"[ACTION] Swipe {direction}")

        # Draw finger tracking on camera preview
        gesture.draw_track(cam_small)
        if fingertip:
            # Scale fingertip to PIP coordinates
            h, w = frame.shape[:2]
            px = int(fingertip[0] * cam_width / w)
            py = int(fingertip[1] * cam_height / h)
            cv2.circle(cam_small, (px, py), 6, (0, 0, 255), -1)

        # Render game board
        game_img = draw_board(game)

        # Overlay camera PIP on bottom-right
        h_game, w_game = game_img.shape[:2]
        pip_x = w_game - cam_width - 15
        pip_y = h_game - cam_height - 15
        game_img[pip_y:pip_y + cam_height, pip_x:pip_x + cam_width] = cam_small

        # Draw PIP border
        cv2.rectangle(game_img, (pip_x - 2, pip_y - 2),
                      (pip_x + cam_width + 2, pip_y + cam_height + 2),
                      (255, 255, 255), 2)

        # Show the game
        cv2.imshow("2048 - Gesture Control", game_img)

        # Handle keyboard
        key = cv2.waitKey(30) & 0xFF
        if key == 27:  # ESC
            break
        elif key == ord('r'):
            game.reset()
            print("[ACTION] Game reset")

    cap.release()
    cv2.destroyAllWindows()
    print("[OK] Program exited")


if __name__ == "__main__":
    main()
```
